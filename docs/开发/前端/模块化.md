## ES6模块化


```javascript
//1. 导入导出：一个变量
//导出(a.js)
let name = 'Tom';
export {name} 
//导入(b.js)
import {name} from './a.js';
console.log(name);


//2. 导入导出：多个变量
//导出(a.js)
let name = 'Tom';
let age = 24;
let gender = 'female';
export {name,age,gender} 
//导入(b.js)
import {name,age,gender} from './a.js'
console.log(name);
console.log(age);
console.log(gender);


//3. 导入导出： 多个函数
//导出(a.js)
let getName = function(){
    return 'Tom';
};
let getAge = function(){
    return 24;
};
let getGender = function(){
    return 'female';
}
export {getName,getAge,getGender} 
//导入(b.js)
import {getName,getAge,getGender} from './a.js'
console.log(getName());
console.log(getAge());
console.log(getGender());


//4. 导入导出：重命名(1)
//导出(a.js)
let name = 'Tom';
export {name as myName} 
//导入(b.js)
import {myName} from './a.js';
console.log(myName);


//5. 导入导出：重命名(2)
//导出(a.js)
let name = 'Tom';
export {name} 
//导入(b.js)
import {name as myName} from './a.js';
console.log(myName);
```
